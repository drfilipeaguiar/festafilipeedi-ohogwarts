<script>
// === DADOS INICIAIS ===
const LIMITE_TOTAL = 60;
const LIMITE_CASA = 15;
const LIMITE_FACCAO = 30;
const ADMIN_PASSWORD = 'hogwarts2027';
const EVENT_DATE = new Date('2027-04-16T18:30:00-03:00'); // Horário de Brasília

let convidados = JSON.parse(localStorage.getItem('convidados')) || [];

function salvar() { localStorage.setItem('convidados', JSON.stringify(convidados)); }

function atualizarPainel() {
  const total = convidados.length;
  document.getElementById('totalVagas').textContent = total;
  const casas = { 'Grifinória':0, 'Sonserina':0, 'Lufa-Lufa':0, 'Corvinal':0 };
  const faccoes = { 'Ordem da Fênix':0, 'Comensais da Morte':0 };
  convidados.forEach(c => {
    if (c.casa && casas[c.casa] !== undefined) casas[c.casa]++;
    if (c.faccao && faccoes[c.faccao] !== undefined) faccoes[c.faccao]++;
  });
  document.getElementById('gryffindorVagas').textContent = casas['Grifinória'];
  document.getElementById('slytherinVagas').textContent = casas['Sonserina'];
  document.getElementById('hufflepuffVagas').textContent = casas['Lufa-Lufa'];
  document.getElementById('ravenclawVagas').textContent = casas['Corvinal'];
  document.getElementById('ordemVagas').textContent = faccoes['Ordem da Fênix'];
  document.getElementById('comensaisVagas').textContent = faccoes['Comensais da Morte'];
  // Destacar esgotados
  document.querySelectorAll('#vagasPanel p').forEach(p => {
    const span = p.querySelector('span');
    if (span) {
      const [atual, total] = span.textContent.split('/');
      if (parseInt(atual) >= parseInt(total)) { span.classList.add('esgotado'); } else { span.classList.remove('esgotado'); }
    }
  });
}

// Função para sortear casa baseada no quiz
function sortearCasa(respostas) {
  const contagem = [0,0,0,0]; // [Grif, Sons, Lufa, Corv]
  respostas.forEach(r => { if (r !== null) contagem[r]++; });
  let max = Math.max(...contagem);
  // Se empate, escolher aleatório entre as maiores
  let indices = contagem.map((v,i) => v === max ? i : -1).filter(v => v !== -1);
  const sorteado = indices[Math.floor(Math.random() * indices.length)];
  const casas = ['Grifinória','Sonserina','Lufa-Lufa','Corvinal'];
  return casas[sorteado];
}

// Contador regressivo
function updateCountdown() {
  const now = new Date();
  const diff = EVENT_DATE - now;
  if (diff <= 0) {
    document.getElementById('countdown').innerHTML = '<h2>O evento já começou!</h2>';
    return;
  }
  const days = Math.floor(diff / (1000*60*60*24));
  const hours = Math.floor((diff % (1000*60*60*24)) / (1000*60*60));
  const minutes = Math.floor((diff % (1000*60*60)) / (1000*60));
  const seconds = Math.floor((diff % (1000*60)) / 1000);
  document.getElementById('days').textContent = String(days).padStart(2,'0');
  document.getElementById('hours').textContent = String(hours).padStart(2,'0');
  document.getElementById('minutes').textContent = String(minutes).padStart(2,'0');
  document.getElementById('seconds').textContent = String(seconds).padStart(2,'0');
}
setInterval(updateCountdown, 1000);
updateCountdown();

// Submissão
const form = document.getElementById('rsvpForm');
form.addEventListener('submit', function(e) {
  e.preventDefault();
  // Coletar dados
  const nome = document.getElementById('nome').value.trim();
  const casaSelect = document.getElementById('casa').value;
  const faccaoSelect = document.getElementById('faccao').value;
  
  // Validar nome duplicado
  if (convidados.some(c => c.nome.toLowerCase() === nome.toLowerCase())) {
    alert('Este nome já está registrado!');
    return;
  }
  
  // Verificar limites
  if (convidados.length >= LIMITE_TOTAL) { alert('Todas as vagas esgotadas!'); return; }
  if (casaSelect) {
    const countCasa = convidados.filter(c => c.casa === casaSelect).length;
    if (countCasa >= LIMITE_CASA) { alert(`Vagas para ${casaSelect} esgotadas!`); return; }
  }
  if (faccaoSelect) {
    const countFaccao = convidados.filter(c => c.faccao === faccaoSelect).length;
    if (countFaccao >= LIMITE_FACCAO) { alert(`Vagas para ${faccaoSelect} esgotadas!`); return; }
  }
  
  // Coletar respostas do quiz
  const respostas = [];
  for (let i = 1; i <= 10; i++) {
    const selected = document.querySelector(`input[name="q${i}"]:checked`);
    respostas.push(selected ? parseInt(selected.value) : null);
  }
  
  const casaSorteada = sortearCasa(respostas);
  
  // Adicionar
  convidados.push({
    nome: nome,
    casa: casaSelect || casaSorteada,
    faccao: faccaoSelect || 'Indefinido'
  });
  salvar();
  atualizarPainel();
  
  // Exibir confirmação
  document.getElementById('rsvpSection').style.display = 'none';
  document.getElementById('confirmation').style.display = 'block';
  document.getElementById('confMsg').textContent = `${nome}, você foi sorteado para a casa ${casaSorteada}!`;
  
  // Animação do selo
  const seal = document.querySelector('.seal');
  seal.style.animation = 'glow 1s infinite alternate';
});

// Admin reset
function showAdmin() {
  const pwd = prompt('Digite a senha de admin:');
  if (pwd === ADMIN_PASSWORD) {
    if (confirm('Tem certeza que deseja resetar todos os dados?')) {
      localStorage.removeItem('convidados');
      convidados = [];
      atualizarPainel();
      document.getElementById('rsvpSection').style.display = 'block';
      document.getElementById('confirmation').style.display = 'none';
      alert('Dados resetados com sucesso!');
    }
  } else {
    alert('Senha incorreta!');
  }
}

// Inicializar painel
atualizarPainel();
// Mostrar painel após 1 segundo
document.addEventListener('DOMContentLoaded', function() {
  document.getElementById('vagasPanel').style.display = 'block';
});
</script>

1. Design & Estética:

Tema 'Dark Wizard': Cores Azul Marinho (#001f3f), Dourado (#d4af37) e Preto (#000).
Fundo: Gradiente radial elegante simulando o pôr do sol na Urca com silhuetas de Hogwarts no rodapé.
Fontes: 'Cinzel' e 'MedievalSharp' (Google Fonts).
Responsividade: Totalmente adaptado para celulares (mobile-first).

2. Conteúdo e Informações:

Título: 'O Grande Baile Bruxo no Rio'.
Contador regressivo para 16/04/2027 às 18:00.
Seção de Localização: Mapa do Flutuante Rio (Urca) e aviso sobre o serviço de translado incluso.
Destaque: 'Chegada às 18:00 para o pôr do sol na Baía de Guanabara'.

3. Lógica do RSVP (Chapéu Seletor):

Formulário com campo de nome e um teste de 10 perguntas psicológicas/não óbvias (dilemas morais e legado).
Gestão de Vagas: 15 vagas confirmadas + 5 vagas de lista de espera por Casa (Grifinória, Sonserina, Corvinal, Lufa-Lufa). Total: 60 confirmados e 20 em espera.
Facções: Atribuição automática (30 Ordem da Fênix / 30 Comensais da Morte) baseada no perfil das respostas.
Persistência: Salvar dados no LocalStorage para evitar nomes duplicados e manter o contador de vagas ativo no navegador.

4. Administração e Confirmação:

Botão discreto de 'Admin' para reset de dados (Senha: hogwarts2027).
Selo animado de confirmação (coruja ou castelo) após o envio bem-sucedido."
